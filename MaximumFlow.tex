% ===================== CODE =====================

\begin{frame}[fragile]{\hlwhite{Code (Core)}}
\begin{lstlisting}[language=C++]
while(bfs()){
    int flow = INF;
    for(int v = t; v != s; v = par[v]){
        int u = par[v];
        flow = min(flow, cap[u][v]);
    }

    for(int v = t; v != s; v = par[v]){
        int u = par[v];
        cap[u][v] -= flow;
        cap[v][u] += flow;
    }

    maxflow += flow;
}
\end{lstlisting}
\end{frame}
