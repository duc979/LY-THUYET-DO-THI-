% ===================== KHÁI NIỆM =====================

\begin{frame}{\hlwhite{Định nghĩa}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Prim là gì?}}]
\hlblue{Prim} là thuật toán tìm \hlgreen{cây khung nhỏ nhất} \hlpurple{(Minimum Spanning Tree - MST)}
trong một đồ thị vô hướng, liên thông, có trọng số.
\end{tcolorbox}

\begin{itemize}
\item Bắt đầu từ một đỉnh bất kỳ
\item Mở rộng dần cây khung bằng cách chọn cạnh nhỏ nhất
\item Không tạo chu trình
\end{itemize}
\end{frame}

% ===================== MST =====================

\begin{frame}{\hlwhite{Cây khung nhỏ nhất là gì?}}
\begin{itemize}
\item \hlblue{Cây khung} là tập cạnh nối tất cả các đỉnh của đồ thị
\item Có đúng $n-1$ cạnh nếu đồ thị có $n$ đỉnh
\item \hlred{Không có chu trình}
\item \hlgreen{Tổng trọng số nhỏ nhất} trong tất cả các cây khung
\end{itemize}

\begin{tcolorbox}
\centering
\hlpurple{Prim xây dựng MST bằng cách mở rộng từ một đỉnh gốc}
\end{tcolorbox}
\end{frame}

% ===================== Ý TƯỞNG =====================

\begin{frame}{\hlwhite{Ý tưởng của Prim}}
\begin{enumerate}
\item Chọn một đỉnh bắt đầu
\item Luôn giữ tập đỉnh đã thuộc MST
\item Từ tập này, chọn cạnh nhỏ nhất nối ra ngoài
\item Thêm đỉnh mới vào MST
\item Lặp lại cho đến khi có đủ $n$ đỉnh
\end{enumerate}

\begin{tcolorbox}
\centering
\hlred{Prim mở rộng cây từng bước từ bên trong ra ngoài}
\end{tcolorbox}
\end{frame}

% ===================== BẢN CHẤT =====================

\begin{frame}{\hlwhite{Bản chất}}
\begin{itemize}
\item Thuật toán \hlblue{Greedy (tham lam)}
\item Mỗi bước chọn cạnh nhỏ nhất hợp lệ
\item Cạnh được chọn phải nối từ cây hiện tại ra một đỉnh chưa thăm
\end{itemize}

\begin{tcolorbox}
\centering
\hlgreen{Mỗi lần thêm một đỉnh mới vào cây khung}
\end{tcolorbox}
\end{frame}

% ===================== CẤU TRÚC DỮ LIỆU =====================

\begin{frame}{\hlwhite{Cấu trúc dữ liệu cần dùng}}
\begin{itemize}
\item Danh sách kề của đồ thị
\item \hlblue{Priority queue / min-heap}
\item Mảng \texttt{visited} để đánh dấu đỉnh đã thuộc MST
\item Mảng \texttt{parent} nếu muốn truy vết cạnh của cây khung
\end{itemize}

\begin{tcolorbox}[title=\textbf{\hlwhite{Vai trò của heap}}]
Heap giúp ta nhanh chóng lấy ra cạnh có trọng số nhỏ nhất đang nối từ cây hiện tại ra ngoài.
\end{tcolorbox}
\end{frame}

% ===================== QUY TRÌNH =====================

\begin{frame}{\hlwhite{Quy trình chi tiết}}
\begin{enumerate}
\item Chọn một đỉnh xuất phát
\item Đưa các cạnh kề của đỉnh đó vào heap
\item Lấy cạnh nhỏ nhất từ heap
\item Nếu đỉnh đầu kia chưa thăm:
\begin{itemize}
\item thêm cạnh vào MST
\item đánh dấu đỉnh mới
\item đưa các cạnh kề của nó vào heap
\end{itemize}
\item Nếu đỉnh đó đã thăm thì bỏ qua
\item Lặp lại đến khi có đủ $n-1$ cạnh
\end{enumerate}
\end{frame}

% ===================== TÍNH CHẤT =====================

\begin{frame}{\hlwhite{Tính chất quan trọng}}
\begin{itemize}
\item \hlblue{Greedy(Tham lam)}: luôn chọn cạnh tốt nhất ở thời điểm hiện tại
\item \hlgreen{An toàn}: cạnh được chọn không làm mất tính tối ưu
\item \hlred{Không tạo chu trình}: vì chỉ nối ra đỉnh chưa thăm
\item Phù hợp với đồ thị có nhiều cạnh kề từ một đỉnh
\end{itemize}
\end{frame}

% ===================== TẠI SAO ĐÚNG =====================

\begin{frame}{\hlwhite{Vì sao Prim đúng?}}
\begin{itemize}
\item Mỗi bước chỉ chọn cạnh nhỏ nhất nối từ cây hiện tại ra ngoài
\item Cạnh đó luôn là lựa chọn an toàn theo tính chất cắt của MST
\item Lặp lại đủ $n-1$ lần sẽ tạo ra một cây khung nhỏ nhất
\end{itemize}

\begin{tcolorbox}
\centering
\hlred{Cốt lõi: mở rộng cây bằng cạnh nhẹ nhất có thể}
\end{tcolorbox}
\end{frame}

\begin{frame}[fragile]{\hlwhite{Code có truy vết cạnh (tiếp)}}
\begin{lstlisting}[language=C++]
    int start = 1;    visited[start] = true;
    for(auto [v, w] : adj[start]) pq.push({w, start, v});
    long long mst_cost = 0;
    vector<tuple<int,int,long long>> mst_edges;
    while(!pq.empty() && (int)mst_edges.size() < n - 1){
        auto [w, u, v] = pq.top();
        pq.pop();
        if(visited[v]) continue;
        visited[v] = true;
        mst_cost += w;
        mst_edges.push_back({u, v, w});
        for(auto [to, w2] : adj[v])
            if(!visited[to]) pq.push({w2, v, to});
    }
    cout << mst_cost << '\n';
    for(auto [u, v, w] : mst_edges)
        cout << u << ' ' << v << ' ' << w << '\n';
    return 0;
}

% ===================== ĐỘ PHỨC TẠP =====================

\begin{frame}{\hlwhite{Độ phức tạp}}
\begin{itemize}
\item Với heap:
\begin{itemize}
\item Mỗi lần push / pop: $O(\log m)$
\item Duyệt cạnh qua danh sách kề
\end{itemize}
\item Tổng độ phức tạp thường dùng:
\end{itemize}

\begin{tcolorbox}
\centering
\hlred{$O(m \log m)$} hoặc \hlred{$O(m \log n)$}
\end{tcolorbox}

\begin{itemize}
\item Tùy cách cài đặt và biểu diễn heap
\end{itemize}
\end{frame}
