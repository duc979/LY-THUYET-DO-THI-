% ===================== KHÁI NIỆM =====================

\begin{frame}{\hlwhite{Định nghĩa}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Khái niệm}}]
\hlblue{Dijkstra} là thuật toán tìm:

\[
\text{Shortest Path từ } s \rightarrow v
\]

trong đồ thị có \hlgreen{trọng số không âm}
\end{tcolorbox}

\begin{itemize}
\item Áp dụng cho \hlred{single-source}
\item Không xử lý được cạnh âm
\end{itemize}
\end{frame}

% ===================== BẢN CHẤT =====================

\begin{frame}{\hlwhite{Bản chất}}
\begin{itemize}
\item Thuật toán \hlblue{Greedy (Tham lam)}
\item Luôn chọn đỉnh có khoảng cách nhỏ nhất
\end{itemize}

\begin{tcolorbox}
\centering
\hlpurple{Một khi đã chọn → khoảng cách là tối ưu}
\end{tcolorbox}
\end{frame}

% ===================== QUY TRÌNH =====================

\begin{frame}{\hlwhite{Quy trình}}
\begin{itemize}
\item Khởi tạo:
\begin{itemize}
\item $dist[s] = 0$
\item còn lại = $\infty$
\end{itemize}

\item Dùng \hlgreen{priority queue}
\item Lặp:
\begin{itemize}
\item Lấy đỉnh nhỏ nhất
\item Relax các cạnh
\end{itemize}
\end{itemize}
\end{frame}

% ===================== RELAX =====================

\begin{frame}{\hlwhite{Relaxation}}
\begin{tcolorbox}
\centering
\[
dist[v] = \min(dist[v], dist[u] + w(u,v))
\]
\end{tcolorbox}

\begin{itemize}
\item Nếu tốt hơn → cập nhật
\item Đẩy lại vào heap
\end{itemize}
\end{frame}

% ===================== CODE =====================

\begin{frame}[fragile]{\hlwhite{Code (Core)}}
\begin{lstlisting}[language=C++]
priority_queue<pair<long long,int>,
    vector<pair<long long,int>>,
    greater<pair<long long,int>>> pq;

dist[s] = 0;
pq.push({0, s});

while(!pq.empty()){
    auto [d, u] = pq.top(); pq.pop();
    if(d > dist[u]) continue;

    for(auto [v, w] : adj[u]){
        if(dist[v] > dist[u] + w){
            dist[v] = dist[u] + w;
            pq.push({dist[v], v});
        }
    }
}
\end{lstlisting}
\end{frame}

% ===================== ĐỘ PHỨC TẠP =====================

\begin{frame}{\hlwhite{Độ phức tạp}}
\begin{itemize}
\item Dùng heap:
\begin{itemize}
\item Push / pop: $\log n$
\end{itemize}
\end{itemize}

\begin{tcolorbox}
\centering
\hlred{$O((n + m)\log n)$}
\end{tcolorbox}
\end{frame}

% ===================== LỖI =====================

\begin{frame}{\hlwhite{Lỗi thường gặp}}
\begin{itemize}
\item \hlred{Có cạnh âm → sai}
\item Không check:
\begin{itemize}
\item \texttt{if(d > dist[u]) continue;}
\end{itemize}
\item Không dùng \hlblue{long long}
\end{itemize}
\end{frame}

% ===================== SO SÁNH =====================

\begin{frame}{\hlwhite{So sánh}}
\begin{itemize}
\item BFS:
\begin{itemize}
\item Trọng số = 1
\end{itemize}

\item Bellman-Ford:
\begin{itemize}
\item Có cạnh âm
\end{itemize}

\item Dijkstra:
\begin{itemize}
\item \hlgreen{Nhanh nhất khi không âm}
\end{itemize}
\end{itemize}
\end{frame}
