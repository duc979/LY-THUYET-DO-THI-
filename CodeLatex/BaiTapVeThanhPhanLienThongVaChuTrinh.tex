\section{Thành phần liên thông}

\begin{frame}{\hlwhite{Bài 1: Đề bài}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Thành phần liên thông}}]
Cho một đồ thị vô hướng gồm $n$ đỉnh và $m$ cạnh.  
Các đỉnh được đánh số từ $1$ đến $n$.

Nhiệm vụ của bạn là:
\begin{itemize}
\item Đếm số \hlblue{thành phần liên thông}
\item In ra các đỉnh thuộc từng thành phần
\end{itemize}
\end{tcolorbox}

\begin{itemize}
\item Đồ thị là \hlgreen{vô hướng}
\item Hai đỉnh thuộc cùng một thành phần nếu giữa chúng tồn tại đường đi
\end{itemize}
\end{frame}

\begin{frame}{\hlwhite{Bài 1: Dữ liệu vào}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Input}}]
Dòng đầu tiên gồm hai số nguyên $n, m$.

$m$ dòng tiếp theo, mỗi dòng gồm hai số nguyên $u, v$ mô tả một cạnh vô hướng nối giữa hai đỉnh $u$ và $v$.
\end{tcolorbox}

\begin{itemize}
\item $1 \le n \le 1000$
\item $0 \le m \le \dfrac{n(n-1)}{2}$
\item $1 \le u, v \le n$
\item Các cạnh không trùng nhau
\end{itemize}
\end{frame}

\begin{frame}{\hlwhite{Bài 1: Dữ liệu ra}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Output}}]
\begin{itemize}
\item Dòng đầu tiên: số thành phần liên thông
\item Mỗi dòng sau: các đỉnh thuộc một thành phần liên thông
\end{itemize}
\end{tcolorbox}

\begin{itemize}
\item Các đỉnh trong cùng một thành phần nên được in theo thứ tự tăng dần
\item Thứ tự các thành phần không bắt buộc, nhưng nên cố định theo cách duyệt
\end{itemize}
\end{frame}

\begin{frame}[fragile]{\hlwhite{Bài 1: Ví dụ}}
\begin{columns}
\column{0.48\textwidth}
\begin{tcolorbox}[title=\textbf{\hlwhite{Sample Input}}]
\begin{lstlisting}[basicstyle=\ttfamily\small]
5 3
1 2
2 3
4 5
\end{lstlisting}
\end{tcolorbox}

\column{0.48\textwidth}
\begin{tcolorbox}[title=\textbf{\hlwhite{Sample Output}}]
\begin{lstlisting}[basicstyle=\ttfamily\small]
2
1 2 3
4 5
\end{lstlisting}
\end{tcolorbox}
\end{columns}
\end{frame}

\begin{frame}{\hlwhite{Bài 1: Giải thích}}
\begin{itemize}
\item Từ đỉnh 1, ta đi được đến 2 và 3 nên tạo thành một thành phần: \{1,2,3\}
\item Từ đỉnh 4, ta đi được đến 5 nên tạo thành thành phần thứ hai: \{4,5\}
\end{itemize}

\begin{tcolorbox}
\centering
\hlpurple{Mỗi lần DFS/BFS từ một đỉnh chưa thăm sẽ tìm ra đúng một thành phần liên thông}
\end{tcolorbox}
\end{frame}

\begin{frame}{\hlwhite{Bài 1: Gợi ý cách làm}}
\begin{itemize}
\item Dùng mảng \texttt{visited[]} để đánh dấu đỉnh đã thăm
\item Duyệt từng đỉnh từ $1$ đến $n$
\item Nếu đỉnh chưa thăm, chạy DFS hoặc BFS từ đỉnh đó
\item Tập các đỉnh đã đi qua trong lần duyệt đó chính là một thành phần liên thông
\end{itemize}

\begin{tcolorbox}
\centering
\hlred{Bài này là ứng dụng trực tiếp của duyệt đồ thị cơ bản}
\end{tcolorbox}
\end{frame}

\begin{frame}{\hlwhite{Bài 1: Subtask}}
\begin{table}[]
\centering
\begin{tabular}{|c|c|c|}
\hline
\rowcolor{blue!10}
\textbf{Subtask} & \textbf{Giới hạn} & \textbf{Điểm} \\
\hline
1 & $n \le 20$ & 30\% \\
\hline
2 & $n \le 200$ & 30\% \\
\hline
3 & $n \le 1000$ & 40\% \\
\hline
\end{tabular}
\end{table}

\begin{tcolorbox}
\centering
\hlgreen{Bài toán có thể giải trong $O(n+m)$ nếu dùng DFS/BFS}
\end{tcolorbox}
\end{frame}

\section{Phát hiện chu trình}

\begin{frame}{\hlwhite{Bài 2: Đề bài}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Phát hiện chu trình}}]
Cho một đồ thị vô hướng gồm $n$ đỉnh và $m$ cạnh.  
Các đỉnh được đánh số từ $1$ đến $n$.

Hãy kiểm tra xem đồ thị có tồn tại \hlblue{chu trình} hay không.
\end{tcolorbox}

\begin{itemize}
\item Nếu có chu trình, in ra \hlred{\texttt{YES}}
\item Nếu không có chu trình, in ra \hlgreen{\texttt{NO}}
\end{itemize}
\end{frame}

\begin{frame}{\hlwhite{Bài 2: Dữ liệu vào}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Input}}]
Dòng đầu tiên gồm hai số nguyên $n, m$.

$m$ dòng tiếp theo, mỗi dòng gồm hai số nguyên $u, v$ mô tả một cạnh vô hướng nối giữa hai đỉnh $u$ và $v$.
\end{tcolorbox}

\begin{itemize}
\item $1 \le n \le 1000$
\item $0 \le m \le \dfrac{n(n-1)}{2}$
\item $1 \le u, v \le n$
\item Các cạnh không trùng nhau
\end{itemize}
\end{frame}

\begin{frame}{\hlwhite{Bài 2: Dữ liệu ra}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Output}}]
\begin{itemize}
\item In ra \texttt{YES} nếu đồ thị có chu trình
\item In ra \texttt{NO} nếu đồ thị không có chu trình
\end{itemize}
\end{tcolorbox}

\begin{itemize}
\item Chỉ cần xác định có tồn tại ít nhất một chu trình hay không
\end{itemize}
\end{frame}

\begin{frame}[fragile]{\hlwhite{Bài 2: Ví dụ}}
\begin{columns}
\column{0.48\textwidth}
\begin{tcolorbox}[title=\textbf{\hlwhite{Sample Input}}]
\begin{lstlisting}[basicstyle=\ttfamily\small]
4 4
1 2
2 3
3 1
3 4
\end{lstlisting}
\end{tcolorbox}

\column{0.48\textwidth}
\begin{tcolorbox}[title=\textbf{\hlwhite{Sample Output}}]
\begin{lstlisting}[basicstyle=\ttfamily\small]
YES
\end{lstlisting}
\end{tcolorbox}
\end{columns}
\end{frame}

\begin{frame}{\hlwhite{Bài 2: Giải thích}}
\begin{itemize}
\item Các đỉnh 1, 2, 3 tạo thành chu trình: $1 \rightarrow 2 \rightarrow 3 \rightarrow 1$
\item Vì tồn tại ít nhất một chu trình nên kết quả là \texttt{YES}
\end{itemize}

\begin{tcolorbox}
\centering
\hlpurple{Một chu trình là một đường đi khép kín, bắt đầu và kết thúc tại cùng một đỉnh}
\end{tcolorbox}
\end{frame}

\begin{frame}{\hlwhite{Bài 2: Gợi ý cách làm}}
\begin{itemize}
\item Dùng DFS để duyệt đồ thị
\item Khi đi từ đỉnh $u$ sang đỉnh $v$:
\begin{itemize}
\item nếu $v$ chưa thăm thì tiếp tục DFS
\item nếu $v$ đã thăm và $v$ khác đỉnh cha của $u$ thì đã phát hiện chu trình
\end{itemize}
\item Có thể dừng ngay khi tìm thấy chu trình đầu tiên
\end{itemize}

\begin{tcolorbox}
\centering
\hlred{Với đồ thị vô hướng, phải bỏ qua cạnh quay về đỉnh cha để tránh báo chu trình sai}
\end{tcolorbox}
\end{frame}
