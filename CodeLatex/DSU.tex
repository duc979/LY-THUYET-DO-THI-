% -------------------- mở đầu --------------------

\begin{frame}{\hlwhite{DSU là gì?}}
\begin{tcolorbox}[title=\textbf{\hlgreen{Khái niệm}}]
\hlblue{DSU} là viết tắt của \hlpurple{Disjoint Set Union}, còn gọi là \hlpurple{Union-Find}. Đây là cấu trúc dữ liệu dùng để quản lý các tập hợp \hlred{rời nhau}.
\end{tcolorbox}

\begin{itemize}
  \item Mỗi phần tử thuộc đúng \hlblue{một tập}.
  \item Có thể kiểm tra hai phần tử có cùng tập hay không.
  \item Có thể gộp hai tập lại thành một tập mới.
\end{itemize}

\begin{tcolorbox}
\centering
\hlorange{DSU rất hữu ích khi xử lý thành phần liên thông và các thao tác gộp nhóm}
\end{tcolorbox}
\end{frame}

% -------------------- ý tưởng --------------------

\begin{frame}{\hlwhite{Ý tưởng trực quan}}
\begin{itemize}
  \item Ban đầu, mỗi đỉnh là một tập riêng biệt.
  \item Khi có quan hệ mới giữa hai đỉnh, ta có thể cần gộp hai tập của chúng.
  \item Nếu hai đỉnh đã ở cùng một tập, việc gộp lại là không cần thiết.
\end{itemize}

\pause

\begin{tcolorbox}[title=\textbf{\hlgreen{Mục tiêu}}]
DSU giúp duy trì các tập rời nhau một cách nhanh chóng, thay vì phải quét lại toàn bộ cấu trúc sau mỗi thao tác.
\end{tcolorbox}
\end{frame}

% -------------------- ứng dụng --------------------

\begin{frame}{\hlwhite{DSU dùng để làm gì?}}
\begin{itemize}
  \item Quản lý \hlblue{thành phần liên thông} khi thêm cạnh.
  \item Kiểm tra xem hai đỉnh có đang cùng một nhóm hay không.
  \item Phát hiện việc nối thêm một cạnh có tạo ra chu trình trong đồ thị vô hướng hay không.
  \item Giải các bài toán gộp nhóm theo kiểu động.
\end{itemize}

\pause

\begin{tcolorbox}
\centering
\hlred{DSU là một trong những cấu trúc dữ liệu cơ bản nhất khi làm việc với đồ thị}
\end{tcolorbox}
\end{frame}

% -------------------- cấu trúc --------------------

\begin{frame}{\hlwhite{Cấu trúc lưu trữ trong DSU}}
\begin{tcolorbox}[title=\textbf{\hlgreen{Hai mảng quan trọng}}]
\begin{itemize}
  \item \hlblue{\texttt{parent[x]}}: cha của đỉnh \texttt{x} trong cây đại diện.
  \item \hlblue{\texttt{sz[x]}} hoặc \hlblue{\texttt{rank[x]}}: kích thước hoặc độ cao tương đối của cây.
\end{itemize}
\end{tcolorbox}

\begin{itemize}
  \item Nếu \hlred{\texttt{parent[x] = x}} thì \texttt{x} là \hlred{gốc}.
  \item Mỗi tập được biểu diễn bởi một cây.
  \item Gốc của cây là đại diện của cả tập.
\end{itemize}

\pause

\begin{tcolorbox}
\centering
\hlorange{Mỗi phần tử chỉ cần biết cha trực tiếp của nó}
\end{tcolorbox}
\end{frame}

\end{itemize}

% -------------------- find --------------------

\begin{frame}{\hlwhite{Hàm \texttt{find(x)}}}
\begin{tcolorbox}[title=\textbf{\hlgreen{Mục đích}}]
\texttt{find(x)} trả về \hlred{đại diện của tập} chứa phần tử \texttt{x}.
\end{tcolorbox}

\begin{itemize}
  \item Nếu \texttt{x} là gốc thì \texttt{find(x) = x}.
  \item Nếu chưa phải gốc, ta đi lên cha của nó cho đến khi gặp gốc.
  \item Kết quả cuối cùng là đại diện của toàn bộ tập chứa \texttt{x}.
\end{itemize}

\pause

\begin{tcolorbox}[title=\textbf{\hlgreen{Trực giác}}]
\hlblue{Tìm gốc} giống như lần theo các mũi tên trong một cây để tìm ra người đứng đầu.
\end{tcolorbox}
\end{frame}

% -------------------- nén đường đi --------------------

\begin{frame}{\hlwhite{Tối ưu hàm \texttt{find(x)}: nén đường đi}}
\begin{tcolorbox}[title=\textbf{\hlgreen{Nén đường đi}}]
Sau khi tìm được gốc, ta \hlred{cho các đỉnh trên đường đi trỏ thẳng về gốc}. Điều này làm cho các lần truy vấn sau nhanh hơn rất nhiều.
\end{tcolorbox}

\begin{itemize}
  \item Lần đầu có thể đi qua nhiều đỉnh.
  \item Sau khi nén đường đi, cây trở nên phẳng hơn.
  \item Đây là một trong hai tối ưu quan trọng nhất của DSU.
\end{itemize}

\pause

\begin{tcolorbox}
\centering
\hlorange{Nén đường đi giúp DSU chạy rất nhanh trong thực tế}
\end{tcolorbox}
\end{frame}

% -------------------- union --------------------

\begin{frame}{\hlwhite{Hàm \texttt{union(a,b)}}}
\begin{tcolorbox}[title=\textbf{\hlgreen{Mục đích}}]
\texttt{union(a,b)} gộp hai tập chứa \texttt{a} và \texttt{b} lại với nhau.
\end{tcolorbox}

\begin{itemize}
  \item Tìm đại diện của hai tập: \(\,u = \texttt{find(a)}\), \(\,v = \texttt{find(b)}\).
  \item Nếu \(u = v\) thì hai phần tử đã cùng một tập.
  \item Nếu khác nhau, nối một gốc vào gốc còn lại.
\end{itemize}

\pause

\begin{tcolorbox}[title=\textbf{\hlgreen{Nguyên tắc}}]
Không nối tùy ý, mà nên nối cây nhỏ vào cây lớn để giảm độ cao của cấu trúc.
\end{tcolorbox}
\end{frame}

% -------------------- union by size/rank --------------------

\begin{frame}{\hlwhite{Tối ưu \texttt{union}: gộp theo kích thước hoặc hạng}}
\begin{itemize}
  \item \hlblue{Gộp theo kích thước}: luôn nối cây nhỏ hơn vào cây lớn hơn.
  \item \hlgreen{Gộp theo hạng}: luôn nối cây có độ cao thấp hơn vào cây cao hơn.
\end{itemize}

\pause

\begin{tcolorbox}[title=\textbf{\hlgreen{Lợi ích}}]
Hai kỹ thuật này giúp cây không bị lệch quá nhiều, từ đó giữ cho DSU rất nhanh.
\end{tcolorbox}

\begin{tcolorbox}
\centering
\hlred{Kết hợp nén đường đi + gộp theo kích thước/hạng là chuẩn tối ưu của DSU}
\end{tcolorbox}
\end{frame}

  \item Nếu khác nhau, chúng nằm ở hai thành phần khác nhau.
\end{itemize}

% -------------------- mã giả --------------------

\begin{frame}[fragile]{\hlwhite{Mã giả DSU}}
\begin{lstlisting}[language=C++]
make_set(x):
    parent[x] = x
    sz[x] = 1
find(x):
    if parent[x] == x:
        return x
    parent[x] = find(parent[x])
    return parent[x]
union(a, b):
    a = find(a)
    b = find(b)
    if a == b:
        return
    if sz[a] < sz[b]:
        swap(a, b)
    parent[b] = a
    sz[a] += sz[b]
\end{lstlisting}
\end{frame}

% -------------------- giải thích mã giả --------------------

\begin{frame}{\hlwhite{Giải thích mã giả}}
\begin{itemize}
  \item \texttt{make\_set}: tạo một tập mới chỉ chứa một phần tử.
  \item \texttt{find}: đi lên gốc và nén đường đi.
  \item \texttt{union}: tìm gốc của hai tập, nếu khác nhau thì gộp lại.
\end{itemize}

\pause

\begin{tcolorbox}[title=\textbf{\hlgreen{Điểm quan trọng}}]
Dòng
\[
\texttt{parent[x] = find(parent[x])}
\]
chính là bước \hlred{nén đường đi}.
\end{tcolorbox}
\end{frame}

% -------------------- ví dụ tuần tự --------------------

\begin{frame}{\hlwhite{Ví dụ tuần tự các thao tác}}
\begin{tcolorbox}[title=\textbf{\hlgreen{Ban đầu}}]
\[
\{1\},\{2\},\{3\},\{4\},\{5\}
\]
\end{tcolorbox}

\begin{enumerate}
  \item \texttt{union(1,2)} \(\rightarrow\) \(\{1,2\},\{3\},\{4\},\{5\}\)
  \item \texttt{union(3,4)} \(\rightarrow\) \(\{1,2\},\{3,4\},\{5\}\)
  \item \texttt{union(2,3)} \(\rightarrow\) \(\{1,2,3,4\},\{5\}\)
  \item \texttt{find(4)} \(\rightarrow\) trả về đại diện của tập chứa 4
\end{enumerate}

\pause

\begin{tcolorbox}
\centering
\hlblue{Qua mỗi bước, số lượng tập giảm dần khi ta gộp các thành phần}
\end{tcolorbox}
\end{frame}

% -------------------- độ phức tạp --------------------

\begin{frame}{\hlwhite{Độ phức tạp}}
\begin{tcolorbox}[title=\textbf{\hlgreen{Khi tối ưu đầy đủ}}]
Với \hlblue{nén đường đi} và \hlblue{gộp theo kích thước/hạng}, DSU hoạt động rất nhanh trong thực tế.
\end{tcolorbox}

\begin{itemize}
  \item \texttt{find}: gần như \(O(1)\) trung bình.
  \item \texttt{union}: gần như \(O(1)\) trung bình.
  \item Phân tích chặt hơn cho kết quả \(O(\alpha(n))\), với \(\alpha(n)\) là hàm nghịch đảo Ackermann.
\end{itemize}

\pause

\begin{tcolorbox}
\centering
\hlpurple{Trong các bài toán thực tế, DSU gần như rất gần hằng số}
\end{tcolorbox}
\end{frame}

% -------------------- ứng dụng --------------------

\begin{frame}{\hlwhite{Ứng dụng của DSU trong lý thuyết đồ thị}}
