% ===================== KHÁI NIỆM =====================

\begin{frame}{\hlwhite{Định nghĩa}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Khái niệm}}]
\hlblue{Shortest Path} là bài toán tìm đường đi:

\[
\text{từ } s \rightarrow v \text{ sao cho tổng trọng số nhỏ nhất}
\]
\end{tcolorbox}

\begin{itemize}
\item Áp dụng cho \hlred{đồ thị có trọng số}
\item Có thể là:
\begin{itemize}
\item Đồ thị có hướng / vô hướng
\end{itemize}
\end{itemize}
\end{frame}

% ===================== PHÂN LOẠI =====================

\begin{frame}{\hlwhite{Phân loại}}
\begin{itemize}
\item Theo nguồn:
\begin{itemize}
\item \hlblue{Single-source} (1 đỉnh)
\item \hlgreen{All-pairs} (mọi cặp đỉnh)
\end{itemize}

\item Theo trọng số:
\begin{itemize}
\item \hlpurple{Không âm}
\item \hlred{Có âm}
\end{itemize}
\end{itemize}

\begin{tcolorbox}
\centering
\hlcyan{Mỗi loại dùng thuật toán khác nhau}
\end{tcolorbox}
\end{frame}

% ===================== THUẬT TOÁN =====================

\begin{frame}{\hlwhite{Các thuật toán chính}}
\begin{itemize}
\item \hlblue{BFS}
\begin{itemize}
\item Trọng số = 1
\end{itemize}

\item \hlgreen{Dijkstra}
\begin{itemize}
\item Trọng số \hlgreen{$\ge 0$}
\end{itemize}

\item \hlred{Bellman-Ford}
\begin{itemize}
\item Có cạnh âm
\end{itemize}

\item \hlpurple{Floyd-Warshall}
\begin{itemize}
\item All-pairs
\end{itemize}
\end{itemize}
\end{frame}

% ===================== SO SÁNH =====================

\begin{frame}{\hlwhite{So sánh}}
\begin{itemize}
\item BFS:
\[
O(n + m)
\]

\item Dijkstra:
\[
O((n + m)\log n)
\]

\item Bellman-Ford:
\[
O(n \cdot m)
\]

\item Floyd:
\[
O(n^3)
\]
\end{itemize}
\end{frame}

% ===================== VẤN ĐỀ =====================

\begin{frame}{\hlwhite{Vấn đề quan trọng}}
\begin{itemize}
\item \hlred{Negative Cycle}
\begin{itemize}
\item Làm bài toán vô nghĩa
\end{itemize}

\item \hlblue{Relaxation}
\begin{itemize}
\item Cập nhật khoảng cách
\end{itemize}
\end{itemize}

\begin{tcolorbox}
\centering
\hlpurple{Ý tưởng cốt lõi của mọi thuật toán}
\end{tcolorbox}
\end{frame}

% ===================== RELAX =====================

\begin{frame}{\hlwhite{Relaxation}}
\begin{tcolorbox}
\centering
\[
dist[v] = \min(dist[v], dist[u] + w(u,v))
\]
\end{tcolorbox}

\begin{itemize}
\item Nếu đi qua $u$ tốt hơn → cập nhật
\item Lặp lại nhiều lần
\end{itemize}
\end{frame}

% ===================== CODE =====================

\begin{frame}[fragile]{\hlwhite{Code (Relax Core)}}
\begin{lstlisting}[language=C++]
if(dist[v] > dist[u] + w){
    dist[v] = dist[u] + w;
}
\end{lstlisting}
\end{frame}

% ===================== CHỌN THUẬT TOÁN =====================

\begin{frame}{\hlwhite{Chọn thuật toán}}
\begin{itemize}
\item Trọng số = 1 → \hlblue{BFS}
\item Không âm → \hlgreen{Dijkstra}
\item Có âm → \hlred{Bellman-Ford}
\item N nhỏ, all-pairs → \hlpurple{Floyd}
\end{itemize}

\begin{tcolorbox}
\centering
\hlcyan{Chọn đúng thuật toán = AC}
\end{tcolorbox}
\end{frame}

\end{itemize}
