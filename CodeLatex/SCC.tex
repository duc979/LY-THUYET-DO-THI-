% ===================== KHÁI NIỆM =====================

\begin{frame}{\hlwhite{Định nghĩa}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Khái niệm}}]
Một \hlblue{Strongly Connected Component (SCC)} là tập các đỉnh sao cho:
\[
\forall u, v \Rightarrow u \leftrightarrow v
\]
(tồn tại đường đi từ $u \rightarrow v$ và $v \rightarrow u$)
\end{tcolorbox}

\begin{itemize}
\item Áp dụng cho \hlred{đồ thị có hướng}
\item Mỗi đỉnh thuộc đúng \hlgreen{1 SCC}
\end{itemize}
\end{frame}

% ===================== BẢN CHẤT =====================

\begin{frame}{\hlwhite{Bản chất}}
\begin{itemize}
\item Gom các đỉnh có thể đi qua lại
\item Xử lý \hlblue{chu trình}
\end{itemize}

\begin{tcolorbox}
\centering
\hlpurple{SCC = nhóm các đỉnh trong cùng chu trình}
\end{tcolorbox}
\end{frame}

% ===================== TÍNH CHẤT =====================

\begin{frame}{\hlwhite{Tính chất}}
\begin{itemize}
\item Gom mỗi SCC thành 1 đỉnh
\item Thu được đồ thị mới:
\end{itemize}

\begin{tcolorbox}
\centering
\hlred{Condensation Graph là DAG}
\end{tcolorbox}

\begin{itemize}
\item Sau đó có thể dùng \hlblue{Topological Sort}
\end{itemize}
\end{frame}

% ===================== TARJAN =====================

\begin{frame}{\hlwhite{Thuật toán Tarjan}}
\begin{itemize}
\item 1 DFS duy nhất
\item Dùng:
\begin{itemize}
\item \hlblue{num[u]}: thời gian thăm
\item \hlblue{low[u]}: giá trị nhỏ nhất
\end{itemize}
\end{itemize}

\begin{tcolorbox}
\centering
\hlpurple{Nhanh + tối ưu hơn}
\end{tcolorbox}
\end{frame}

\begin{frame}[fragile]{\hlwhite{Code Tarjan (Core)}}
\begin{lstlisting}[language=C++]
void dfs(int u){
    num[u] = low[u] = ++timeDfs;
    st.push(u);
    inStack[u] = true;

    for(int v: adj[u]){
        if(!num[v]){
            dfs(v);
            low[u] = min(low[u], low[v]);
        }
        else if(inStack[v])
            low[u] = min(low[u], num[v]);
    }

    if(low[u] == num[u]){
        while(true){
            int v = st.top(); st.pop();
            inStack[v] = false;
            if(v == u) break;
        }
    }
}
\end{lstlisting}
\end{frame}
