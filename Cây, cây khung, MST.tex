% -------------------- Định nghĩa cây --------------------

\begin{frame}{\hlwhite{Định nghĩa Cây}}
\begin{tcolorbox}[title=\textbf{\hlgreen{Khái niệm}}]
\hlblue{Cây} là một đồ thị vô hướng \emph{\hlgreen{liên thông}} và \emph{\hlred{không chứa chu trình}}.
\end{tcolorbox}

\begin{tcolorbox}
\centering
\hlpurple{Cây = \hlgreen{liên thông} + \hlred{không có chu trình}}
\end{tcolorbox}
\end{frame}

% -------------------- Tính chất --------------------

\begin{frame}{\hlwhite{Tính chất của cây}}
\begin{itemize}
  \item Với \hlblue{$n$ đỉnh} thì một cây có chính xác \hlred{$n-1$ cạnh}.
  \item Giữa hai đỉnh bất kỳ luôn tồn tại \hlgreen{một và chỉ một đường đi}.
  \item Thêm một cạnh bất kỳ vào cây sẽ tạo thành \hlred{chu trình}.
  \item Xóa một cạnh bất kỳ khỏi cây sẽ khiến đồ thị \hlorange{không còn liên thông}.
\end{itemize}
\end{frame}

% -------------------- Ví dụ cây --------------------

\begin{frame}{\hlwhite{Ví dụ minh họa một cây}}
\centering
\begin{tikzpicture}[scale=1]
\tikzset{v/.style={circle,draw=myblue,fill=myblue!10,minimum size=8mm}}

\node[v] (1) at (0,0) {1};

\draw (2)--(5);
\end{tikzpicture}

\vspace{0.3cm}
\hlpurple{Đồ thị trên liên thông, không có chu trình → là cây.}
\end{frame}

% -------------------- Định nghĩa cây khung --------------------

\begin{frame}{\hlwhite{Cây khung (Spanning Tree)}}
\begin{tcolorbox}[title=\textbf{\hlgreen{Định nghĩa}}]
Trong một đồ thị vô hướng $G$, \hlblue{cây khung} là một cây con chứa \hlred{tất cả các đỉnh} của $G$ và là \hlgreen{liên thông} (không chu trình).
\end{tcolorbox}

\begin{itemize}
  \item Mỗi cây khung có đúng \hlred{$n-1$ cạnh} nếu đồ thị có \hlblue{$n$ đỉnh}.
  \item Một đồ thị \hlgreen{liên thông} luôn có ít nhất một cây khung.
\end{itemize}
\end{frame}

% -------------------- Ví dụ cây khung --------------------

\begin{frame}{\hlwhite{Ví dụ cây khung}}
\centering
\begin{tikzpicture}[scale=1]
\tikzset{v/.style={circle,draw=myblue,fill=myblue!10,minimum size=8mm}}
\node[v] (1) at (0,0) {1};
\node[v] (2) at (2,1) {2};
\node[v] (3) at (2,-1) {3};
\node[v] (4) at (4,0) {4};
\draw (1)--(2);
\draw (1)--(3);
\draw (2)--(4);
\draw[dashed] (3)--(4); % cạnh dư
\end{tikzpicture}

\vspace{0.25cm}
\hlpurple{Nếu bỏ cạnh nét đứt, ta thu được một cây khung (liên thông, không chu trình).}
\end{frame}

% -------------------- Số lượng cây khung --------------------

\begin{frame}{\hlwhite{Số lượng và lựa chọn cây khung}}
\begin{itemize}
  \item Một đồ thị có thể có \hlpurple{nhiều cây khung khác nhau}.
  \item Ví dụ đồ thị đầy đủ \hlblue{$K_n$} có số lượng cây khung rất lớn.
  \item Trong nhiều bài toán ta cần chọn cây khung \hlred{"tốt nhất"} theo tiêu chí nào đó (ví dụ \hlorange{tổng chi phí nhỏ nhất}).
\end{itemize}

\begin{tcolorbox}
\centering
\hlblue{Bài toán thực tế: chọn cây khung tối ưu theo chi phí}
\end{tcolorbox}
\end{frame}

% -------------------- MST --------------------

\begin{frame}{\hlwhite{Cây khung nhỏ nhất (MST)}}
\begin{tcolorbox}[title=\textbf{\hlgreen{Khái niệm}}]
Với một đồ thị vô hướng có \hlpurple{trọng số} trên các cạnh, \hlblue{cây khung nhỏ nhất (MST)} là cây khung có \hlred{tổng trọng số các cạnh nhỏ nhất}.
\end{tcolorbox}

\begin{itemize}
  \item MST áp dụng cho các bài toán \hlorange{tối ưu chi phí kết nối}.
  \item MST giữ \hlgreen{liên thông} và \hlred{tối thiểu hóa tổng chi phí}.
\end{itemize}
\end{frame}

% -------------------- Ví dụ MST --------------------

\begin{frame}{\hlwhite{Ví dụ minh họa MST}}
\centering
\begin{tikzpicture}[scale=1]
\tikzset{v/.style={circle,draw=myblue,fill=myblue!10,minimum size=8mm}}

\node[v] (1) at (0,0) {1};
\node[v] (2) at (2,1) {2};
\node[v] (3) at (2,-1) {3};

% các cạnh thuộc MST (nổi bật)
\draw[very thick,myred] (2)-- (4);
\draw[very thick,myred] (1)-- (2);
\draw[very thick,myred] (1)-- (3);

\end{tikzpicture}

\vspace{0.25cm}
\hlpurple{Chọn tập cạnh liên thông sao cho tổng trọng số (2+1+3 = 6) là nhỏ nhất (ví dụ minh họa).}
\end{frame}

% -------------------- Tính chất MST --------------------

\begin{frame}{\hlwhite{Tính chất quan trọng của MST}}
\begin{itemize}
  \item Mỗi MST có đúng \hlred{$n-1$ cạnh} (với \hlblue{$n$ đỉnh}).
  \item Nếu tất cả trọng số \hlgreen{phân biệt} (không trùng nhau) → MST là \hlred{duy nhất}.
  \item Nếu có trọng số \hlorange{bằng nhau} → có thể tồn tại \hlpurple{nhiều MST khác nhau}.
\end{itemize}
\end{frame}

% -------------------- Thuộc tính cut & cycle (tóm tắt) --------------------

\begin{frame}{\hlwhite{Thuộc tính cắt và chu trình (tóm tắt)}}
\begin{itemize}
  \item \textbf{\hlgreen{Cut property (tính chất cắt):}} Trong mọi phân chia tập đỉnh thành hai phần, cạnh có \hlred{trọng số nhỏ nhất} nối hai phần đó luôn thuộc ít nhất một MST.
  \item \textbf{\hlpurple{Cycle property (tính chất chu trình):}} Trong một chu trình, cạnh có \hlred{trọng số lớn nhất} không thể thuộc MST (nếu có cạnh lớn hơn thay thế được).
\end{itemize}
\end{frame}

% -------------------- Ứng dụng --------------------

\begin{frame}{\hlwhite{Ứng dụng của MST}}
\begin{itemize}
  \item Thiết kế \hlblue{mạng đường truyền}, \hlgreen{mạng điện} với chi phí tối thiểu.
  \item Lập kế hoạch xây dựng \hlorange{đường giao thông} với chi phí tiết kiệm.
  \item Ứng dụng trong \hlpurple{phân cụm (clustering)} trong học máy.
  \item \hlcyan{Tiền xử lý} cho các bài toán tối ưu khác trên đồ thị.
\end{itemize}
\end{frame}

% -------------------- So sánh tổng quan --------------------

\begin{frame}{\hlwhite{So sánh ngắn}}
\begin{tcolorbox}
\begin{itemize}
  \item \hlgreen{Cây}: cấu trúc đồ thị liên thông, không chu trình.
  \item \hlgreen{Cây khung}: cây con bao phủ toàn bộ đỉnh của đồ thị.
  \item \hlgreen{MST}: cây khung có tổng trọng số cạnh nhỏ nhất.
\end{itemize}
\end{tcolorbox}
\end{frame}

% -------------------- Kết luận --------------------

\begin{frame}{\hlwhite{Kết luận}}
\begin{itemize}
  \item Cây và cây khung là khái niệm nền tảng trong lý thuyết đồ thị.
  \item MST là bài toán tối ưu ứng dụng rộng rãi trong thực tế.
  \item Các tính chất cut/cycle là nền tảng cho các thuật toán xây MST (Kruskal, Prim — có thể bổ sung sau).
\end{itemize}

\centerline{\hlorange{Cảm ơn!}}
\end{frame}
