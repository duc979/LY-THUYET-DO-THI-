% 1 Definition
\begin{frame}{\hlwhite{Khái niệm chu trình}}
\begin{block}{\hlgreen{Định nghĩa}}
Một \hlblue{chu trình} là đường đi:
\begin{itemize}
\item Bắt đầu và kết thúc tại cùng một đỉnh
\item Không lặp cạnh (và thường không lặp đỉnh)
\end{itemize}
\end{block}

\medskip
\[
\hlpurple{v_1 \rightarrow v_2 \rightarrow ... \rightarrow v_1}
\]

\end{frame}

% 3 Idea
\begin{frame}{\hlwhite{Ý tưởng phát hiện chu trình}}
\begin{itemize}
\item Dùng \hlblue{DFS}
\item Khi duyệt cạnh (u, v):
\begin{itemize}
\item Nếu v chưa thăm → tiếp tục DFS
\item Nếu v đã thăm và \hlred{v != cha của u} → có chu trình
\end{itemize}
\end{itemize}

\medskip
\textbf{Lưu ý:}
\begin{itemize}
\item Đồ thị vô hướng cần kiểm tra cha
\item Đồ thị có hướng dùng trạng thái (visited, in-stack)
\end{itemize}
\end{frame}

% 5 DFS Code
\begin{frame}[fragile]{\hlwhite{Code DFS (vô hướng)}}
\begin{lstlisting}[language=C++]
vector<int> adj[N];
bool vis[N];

bool dfs(int u, int parent){
  vis[u]=true;
  for(int v: adj[u]){
    if(!vis[v]){
      if(dfs(v,u)) return true;
    }
    else if(v!=parent){
      return true;
    }
  }
  return false;
}
\end{lstlisting}
\end{frame}

% 6 Directed Graph
\begin{frame}[fragile]{\hlwhite{Code (có hướng)}}
\begin{lstlisting}[language=C++]
bool vis[N], instack[N];

bool dfs(int u){
  vis[u]=instack[u]=true;
  for(int v: adj[u]){
    if(!vis[v] && dfs(v)) return true;
    else if(instack[v]) return true;
  }
  instack[u]=false;
  return false;
}
\end{lstlisting}
\end{frame}
