% ===================== KHÁI NIỆM =====================

\begin{frame}{\hlwhite{Định nghĩa}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Khái niệm}}]
Topological Sort là cách sắp xếp các đỉnh sao cho:
\[
\hlblue{u \rightarrow v \Rightarrow u \text{ đứng trước } v}
\]
\end{tcolorbox}

\begin{itemize}
\item Áp dụng cho \hlred{đồ thị có hướng}
\item Chỉ tồn tại nếu \hlred{không có chu trình}
\end{itemize}
\end{frame}

% ===================== BẢN CHẤT =====================

\begin{frame}{\hlwhite{Bản chất}}
\begin{itemize}
\item Biểu diễn quan hệ \hlblue{phụ thuộc}
\item Thứ tự xử lý hợp lệ
\item Không có vòng lặp
\end{itemize}

\begin{tcolorbox}
\centering
\hlpurple{Topo = thứ tự hợp lệ của DAG}
\end{tcolorbox}
\end{frame}

% ===================== ĐIỀU KIỆN =====================

\begin{frame}{\hlwhite{Điều kiện tồn tại}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Định lý}}]
Topo Sort tồn tại \hlred{khi và chỉ khi} đồ thị là DAG
\end{tcolorbox}

\begin{itemize}
\item Có chu trình $\Rightarrow$ không thể topo
\item Không chu trình $\Rightarrow$ luôn tồn tại topo
\end{itemize}
\end{frame}

% ===================== DFS =====================

\begin{frame}{\hlwhite{Topo bằng DFS}}
\begin{itemize}
\item DFS toàn bộ đồ thị
\item Sau khi duyệt xong:
\begin{itemize}
\item Đưa vào stack
\end{itemize}
\item Đảo stack → topo
\end{itemize}
\end{frame}

\begin{frame}[fragile]{\hlwhite{Code DFS}}
\begin{lstlisting}[language=C++]
stack<int> st;
bool vis[N];

void dfs(int u){
 vis[u]=true;
 for(int v: adj[u]){
  if(!vis[v]) dfs(v);
 }
 st.push(u);
}
\end{lstlisting}
\end{frame}

% ===================== VÍ DỤ 1 =====================

\begin{frame}{\hlwhite{Ví dụ 1 - Sắp xếp TOPO}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Đề bài}}]
Cho đồ thị có hướng không chu trình (DAG) $G(V, E)$.  
Hãy đánh số lại các đỉnh sao cho với mỗi cạnh $(u, v)$:
\[
u \rightarrow v \Rightarrow u < v
\]
\end{tcolorbox}

\begin{itemize}
\item Input:
\begin{itemize}
\item $n, m$ là số đỉnh và số cạnh
\item $m$ cạnh có dạng $u \rightarrow v$
\end{itemize}

\item Output:
\begin{itemize}
\item In ra thứ tự đánh số lại các đỉnh
\end{itemize}

\item Ý tưởng:
\begin{itemize}
\item \hlblue{Topological Sort}
\item Dùng \hlgreen{Kahn (BFS)}
\end{itemize}

\item Độ phức tạp:
\[
\hlred{O(n + m)}
\]
\end{itemize}
\end{frame}
