% 5 Code DFS
\begin{frame}[fragile]{\hlwhile{Code DFS}}
\begin{lstlisting}[language=C++]
vector<int> adj[N];
bool vis[N];

void dfs(int u){
  vis[u]=true;
  for(int v: adj[u]){
    if(!vis[v]) dfs(v);
  }
}

int count_cc(int n){
  int cnt=0;
  for(int i=1;i<=n;i++){
    if(!vis[i]){
      dfs(i);
      cnt++;
    }
  }
  return cnt;
}
\end{lstlisting}
\end{frame}

% 6 BFS version
\begin{frame}[fragile]{\hlwhite{Code BFS}}
\begin{lstlisting}[language=C++]
void bfs(int s){
  queue<int> q;
  q.push(s);
  vis[s]=true;

  while(!q.empty()){
    int u=q.front(); q.pop();
    for(int v: adj[u]){
      if(!vis[v]){
        vis[v]=true;
        q.push(v);
      }
    }
  }
}
\end{lstlisting}
\end{frame}
