% ===================== ĐỊNH NGHĨA =====================

\begin{frame}{\hlwhite{Đồ thị là gì?}}
\begin{tcolorbox}[title=\textbf{\hlgreen{Định nghĩa}}]
Đồ thị là cấu trúc gồm các \hlblue{đỉnh (vertices)} và \hlblue{cạnh (edges)} biểu diễn mối quan hệ.
\end{tcolorbox}

\begin{tcolorbox}
\centering
\hlpurple{$G = (V, E)$}
\end{tcolorbox}

\begin{itemize}
\item $V$: tập đỉnh
\item $E$: tập cạnh
\end{itemize}
\end{frame}

% ===================== MINH HỌA =====================

\begin{frame}{\hlwhite{Ví dụ đồ thị}}
\centering
\begin{tikzpicture}[scale=1]
\tikzset{v/.style={circle,draw=myblue,fill=myblue!10}}

\node[v] (A) at (0,0) {A};
\node[v] (B) at (2,0) {B};
\node[v] (C) at (1,2) {C};

\draw (A) -- (B);
\draw (B) -- (C);
\draw (A) -- (C);
\end{tikzpicture}
\end{frame}

% ===================== PHÂN LOẠI =====================

\begin{frame}{\hlwhite{Phân loại đồ thị}}
\begin{itemize}
\item \hlblue{Đồ thị vô hướng}
\item \hlblue{Đồ thị có hướng}
\item \hlblue{Đồ thị có trọng số}
\end{itemize}

\begin{tcolorbox}
\centering
\hlpurple{$(u,v) \neq (v,u)$ trong đồ thị có hướng}
\end{tcolorbox}
\end{frame}

% ===================== BẬC ĐỈNH =====================

\begin{frame}{\hlwhite{Bậc của đỉnh}}
\begin{itemize}
\item Đồ thị vô hướng:
\[
deg(v) = \text{số cạnh nối với } v
\]

\item Đồ thị có hướng:
\begin{itemize}
\item In-degree
\item Out-degree
\end{itemize}
\end{itemize}
\end{frame}

% ===================== PATH & CYCLE =====================

\begin{frame}{\hlwhite{Đường đi và Chu trình}}
\begin{tcolorbox}
\centering
\hlpurple{$v_1 \rightarrow v_2 \rightarrow \dots \rightarrow v_k$}
\end{tcolorbox}

\begin{itemize}
\item \hlblue{Path}: dãy các đỉnh liên tiếp
\item \hlred{Cycle}: quay lại đỉnh ban đầu
\end{itemize}
\end{frame}

\item Liên thông yếu
\end{itemize}
\end{itemize}
