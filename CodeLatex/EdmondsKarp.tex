\begin{frame}[fragile]{\hlwhite{Code (Core)}}
\begin{lstlisting}[language=C++]
bool bfs(){
    memset(par, -1, sizeof(par));
    queue<int> q;
    q.push(s);
    par[s] = s;

    while(!q.empty()){
        int u = q.front(); q.pop();
        for(int v : adj[u]){
            if(par[v] == -1 && cap[u][v] > 0){
                par[v] = u;
                if(v == t) return true;
                q.push(v);
            }
        }
    }
    return false;
}

while(bfs()){
    int flow = INF;

    for(int v = t; v != s; v = par[v]){
        int u = par[v];
        flow = min(flow, cap[u][v]);
    }

    for(int v = t; v != s; v = par[v]){
        int u = par[v];
        cap[u][v] -= flow;
        cap[v][u] += flow;
    }

    maxflow += flow;
}
\end{lstlisting}
\end{frame}
