% ===================== KHÁI NIỆM =====================

\begin{frame}{\hlwhite{Định nghĩa}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Ý tưởng}}]
\hlblue{Ford–Fulkerson} là thuật toán tìm \hlred{maximum flow} bằng cách:

\[
\text{Liên tục tìm augmenting path từ } s \rightarrow t
\]
\end{tcolorbox}

\begin{itemize}
\item Dựa trên \hlgreen{residual graph}
\item Có thể dùng DFS/BFS
\end{itemize}
\end{frame}

% ===================== AUGMENTING =====================

\begin{frame}{\hlwhite{Augmenting Path}}
\begin{itemize}
\item Là đường từ $s$ đến $t$ còn capacity
\item Flow tăng theo:
\[
\min \text{ capacity trên đường}
\]
\end{itemize}

\begin{tcolorbox}
\centering
\hlpurple{Nút thắt cổ chai (bottleneck)}
\end{tcolorbox}
\end{frame}

% ===================== RESIDUAL =====================

\begin{frame}{\hlwhite{Residual Graph}}
\begin{itemize}
\item Cạnh $(u,v)$:
\begin{itemize}
\item $cap[u][v]$: còn lại
\item $cap[v][u]$: luồng ngược
\end{itemize}
\end{itemize}

\begin{tcolorbox}
\centering
\hlred{Cho phép "quay lui" luồng}
\end{tcolorbox}
\end{frame}

% ===================== THUẬT TOÁN =====================

\begin{frame}{\hlwhite{Thuật toán}}
\begin{itemize}
\item Lặp:
\begin{itemize}
\item Tìm đường $s \rightarrow t$
\item Cập nhật flow
\end{itemize}

\item Dừng khi:
\begin{itemize}
\item Không còn đường tăng luồng
\end{itemize}
\end{itemize}

\begin{tcolorbox}
\centering
\hlcyan{Greedy + DFS/BFS}
\end{tcolorbox}
\end{frame}

% ===================== CODE =====================

\begin{frame}[fragile]
\frametitle{\hlwhite{Code (Core)}}
\begin{lstlisting}[language=C++, breaklines=true]
bool dfs(int u, int flow){
    if(u == t) return flow;
    vis[u] = true;
    for(int v : adj[u]){
        if(!vis[v] && cap[u][v] > 0){
            int pushed = dfs(v, min(flow, cap[u][v]));
            if(pushed){
                cap[u][v] -= pushed;
                cap[v][u] += pushed;
                return pushed;
            }
        }
    }
    return 0;
}
while(int pushed = dfs(s, INF)){
    maxflow += pushed;
}
\end{lstlisting}
}
\end{frame}
