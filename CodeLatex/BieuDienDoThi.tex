\section{Khái niệm cơ bản}
\begin{frame}{Khái niệm cơ bản}
Đồ thị gồm:
\begin{itemize}
    \item Tập đỉnh $V$
    \item Tập cạnh $E$
\end{itemize}

Trong C++, đồ thị thường được lưu bằng:
\begin{itemize}
    \item Danh sách kề (Adjacency List)
    \item Ma trận kề (Adjacency Matrix)
    \item Danh sách cạnh (Edge List)
\end{itemize}
\end{frame}

\section{Danh sách kề}
\begin{frame}{Adjacency List}
Mỗi đỉnh lưu danh sách các đỉnh kề với nó.

\textbf{Cài đặt:}
\begin{itemize}
    \item vector<int> adj[N];
\end{itemize}

\textbf{Ví dụ cạnh:}
\[
(1,2), (1,3)
\]

\textbf{Lưu:}
\begin{itemize}
    \item adj[1] = \{2,3\}
\end{itemize}
\end{frame}

\section{Ma trận kề}
\begin{frame}{Adjacency Matrix}
Dùng mảng 2 chiều $a[i][j]$

\begin{itemize}
    \item $a[i][j] = 1$ nếu có cạnh
    \item $a[i][j] = 0$ nếu không có
\end{itemize}

\textbf{Ví dụ:}

\small
\[
\begin{array}{|c|c|c|c|}
\hline
 & 1 & 2 & 3 \\
\hline
1 & 0 & 1 & 1 \\
2 & 1 & 0 & 0 \\
3 & 1 & 0 & 0 \\
\hline
\end{array}
\]
\end{frame}

\section{Danh sách cạnh}
\begin{frame}{Edge List}
Lưu trực tiếp các cạnh.

\textbf{Dạng:}
\[
(u, v)
\]

\textbf{Ví dụ:}
\begin{itemize}
    \item (1,2)
    \item (1,3)
    \item (2,3)
\end{itemize}

Thường dùng khi:
\begin{itemize}
    \item Thuật toán Kruskal
\end{itemize}
\end{frame}

\section{So sánh}
\begin{frame}{So sánh}
\begin{itemize}
\item Danh sách kề:
    \begin{itemize}
        \item Tiết kiệm bộ nhớ
        \item Duyệt nhanh
    \end{itemize}

    \item Ma trận kề:
    \begin{itemize}
        \item Kiểm tra cạnh nhanh
        \item Tốn bộ nhớ
    \end{itemize}

    \item Danh sách cạnh:
    \begin{itemize}
        \item Dễ lưu
        \item Phù hợp thuật toán MST
    \end{itemize}
\end{itemize}
\end{frame}
