% ===================== KHÁI NIỆM =====================

\begin{frame}{\hlwhite{Định nghĩa}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Kruskal là gì?}}]
\hlblue{Kruskal} là thuật toán tìm \hlgreen{cây khung nhỏ nhất} \hlpurple{(Minimum Spanning Tree - MST)}
trong một đồ thị vô hướng, liên thông, có trọng số.
\end{tcolorbox}

\begin{itemize}
\item Mục tiêu: nối tất cả các đỉnh với tổng trọng số nhỏ nhất
\item Chọn các cạnh \hlred{nhỏ nhất trước}
\item Không được tạo chu trình
\end{itemize}
\end{frame}

% ===================== MST LÀ GÌ =====================

\begin{frame}{\hlwhite{Cây khung nhỏ nhất là gì?}}
\begin{itemize}
\item \hlblue{Cây khung} là tập cạnh nối tất cả các đỉnh của đồ thị
\item Có đúng $n-1$ cạnh nếu đồ thị có $n$ đỉnh
\item \hlred{Không có chu trình}
\item \hlgreen{Tổng trọng số nhỏ nhất} trong tất cả các cây khung
\end{itemize}

\begin{tcolorbox}
\centering
\hlpurple{Kruskal giúp ta xây dựng MST bằng cách chọn cạnh tốt nhất từng bước}
\end{tcolorbox}
\end{frame}

% ===================== Ý TƯỞNG =====================

\begin{frame}{\hlwhite{Ý tưởng của Kruskal}}
\begin{enumerate}
\item Sắp xếp tất cả các cạnh theo trọng số tăng dần
\item Duyệt từng cạnh từ nhỏ đến lớn
\item Nếu cạnh đó \hlgreen{không tạo chu trình} thì chọn
\item Nếu tạo chu trình thì bỏ qua
\item Dừng khi đã chọn đủ $n-1$ cạnh
\end{enumerate}

\begin{tcolorbox}
\centering
\hlred{Chọn cạnh nhỏ nhất nhưng vẫn phải giữ đồ thị là một cây}
\end{tcolorbox}
\end{frame}

% ===================== CẤU TRÚC DỮ LIỆU =====================

\begin{frame}{\hlwhite{Cấu trúc dữ liệu cần dùng}}
\begin{itemize}
\item Danh sách cạnh
\item Sắp xếp cạnh theo trọng số
\item \hlblue{Disjoint Set Union (DSU) / Union-Find}
\end{itemize}

\begin{tcolorbox}[title=\textbf{\hlwhite{Vai trò của DSU}}]
DSU dùng để kiểm tra nhanh hai đỉnh đang thuộc cùng một thành phần liên thông hay chưa.
Nếu đã cùng thành phần, thêm cạnh vào sẽ tạo chu trình.
\end{tcolorbox}
\end{frame}

% ===================== DSU =====================

\begin{frame}{\hlwhite{Disjoint Set Union}}
\begin{itemize}
\item \texttt{find(x)}: tìm gốc của tập chứa đỉnh $x$
\item \texttt{union(a,b)}: gộp hai tập lại với nhau
\item Dùng \hlgreen{path compression} và \hlgreen{union by size/rank}
\end{itemize}

\begin{tcolorbox}
\centering
\[
\text{Nếu } find(u) \neq find(v) \Rightarrow \text{có thể chọn cạnh } (u,v)
\]
\end{tcolorbox}
\end{frame}

% ===================== QUY TRÌNH CHI TIẾT =====================

\begin{frame}{\hlwhite{Quy trình chi tiết}}
\begin{enumerate}
\item Khởi tạo mỗi đỉnh là một tập riêng
\item Sắp xếp cạnh theo trọng số tăng dần

% ===================== TẠI SAO ĐÚNG =====================

\begin{frame}{\hlwhite{Vì sao Kruskal đúng?}}
\begin{itemize}
\item Khi xét cạnh theo thứ tự tăng dần, cạnh nhỏ hơn luôn được ưu tiên
\item Nếu cạnh nối hai thành phần khác nhau, nó không thể làm xuất hiện chu trình
\item Việc chọn cạnh nhỏ nhất an toàn giúp tổng trọng số toàn cây là nhỏ nhất
\end{itemize}

\begin{tcolorbox}
\centering
\hlred{Ý tưởng cốt lõi: chọn cạnh nhẹ nhất mà vẫn giữ được tính cây}
\end{tcolorbox}
\end{frame}

% ===================== CODE =====================

\begin{frame}[fragile]{\hlwhite{Code C++: Cài đặt trước}}
\begin{lstlisting}[language=C++]
#include <bits/stdc++.h>
using namespace std;
struct Edge{
    int u, v;
    long long w;
};
int n, m, parent[MAXN], sz[MAXN];
vector<Edge> edges(m);
const int MAXN = 100005;
int find_set(int u){
    if(parent[u] == u) return u;
    return parent[u] = find_set(parent[u]);
}
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]{\hlwhite{Code C++: Hàm kiểm tra}}
\begin{lstlisting}[language=C++]
bool unite(int u, int v){
    u = find_set(u);
    v = find_set(v);
    if(u == v) return false;
    if(sz[u] < sz[v]) swap(u, v);
    parent[v] = u;
    sz[u] += sz[v];
    return true;
}
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]{\hlwhite{Code C++: Khai báo}}
\begin{lstlisting}[language=C++]
void khaibao(){
    cin >> n >> m;
    for(int i = 0; i < m; i++)
        cin >> edges[i].u >> edges[i].v >> edges[i].w;
    sort(edges.begin(), edges.end(), [](const Edge &a, const Edge &b){
        return a.w < b.w;
    });
    for(int i = 1; i <= n; i++){
        parent[i] = i;
        sz[i] = 1;
    }
}
\end{lstlisting}
\end{frame}
\begin{frame}[fragile]{\hlwhite{Code C++: Trong hàm chính}}
\begin{lstlisting}[language=C++]
int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    long long mst_cost = 0;
    vector<Edge> chosen;
    for(auto &e : edges)
        if(unite(e.u, e.v)){
            mst_cost += e.w;
            chosen.push_back(e);
        }
    cout << mst_cost << '\n';
    for(auto &e : chosen)
        cout<<e.u <<' '<<e.v<<' '<<e.w<<'\n';
}
\end{lstlisting}
\end{frame}

% ===================== ĐỘ PHỨC TẠP =====================

\begin{frame}{\hlwhite{Độ phức tạp}}
\begin{itemize}
\item Sắp xếp $m$ cạnh: $O(m \log m)$
\item DSU:
\begin{itemize}
\item \texttt{find}, \texttt{union} gần như $O(1)$ trung bình
\end{itemize}
\end{itemize}

\begin{tcolorbox}
\centering
\hlred{Tổng độ phức tạp: $O(m \log m)$}
\end{tcolorbox}
\end{frame}
