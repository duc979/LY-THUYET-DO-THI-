% ===================== KHÁI NIỆM =====================

\begin{frame}{\hlwhite{Định nghĩa}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Khái niệm}}]
\hlblue{All-pairs Shortest Paths (APSP)} là bài toán tìm:

\[
\text{Shortest Path giữa mọi cặp } (u, v)
\]
\end{tcolorbox}

\begin{itemize}
\item Khác với \hlgreen{single-source}
\item Trả về ma trận khoảng cách
\end{itemize}
\end{frame}

% ===================== CÁCH GIẢI =====================

\begin{frame}{\hlwhite{Cách giải}}
\begin{itemize}
\item Chạy nhiều lần:
\begin{itemize}
\item \hlblue{Dijkstra} (n lần)
\item \hlred{Bellman-Ford} (n lần)
\end{itemize}

\item Hoặc:
\begin{itemize}
\item \hlpurple{Floyd-Warshall}
\end{itemize}
\end{itemize}

\begin{tcolorbox}
\centering
\hlcyan{Tuỳ bài mà chọn}
\end{tcolorbox}
\end{frame}

% ===================== FLOYD =====================

\begin{frame}{\hlwhite{Floyd-Warshall}}
\begin{itemize}
\item Ý tưởng:
\begin{itemize}
\item Xét đỉnh trung gian $k$
\end{itemize}
\end{itemize}

\begin{tcolorbox}
\centering
\[
dist[i][j] = \min(dist[i][j], dist[i][k] + dist[k][j])
\]
\end{tcolorbox}
\end{frame}

% ===================== THUẬT TOÁN =====================

\begin{frame}{\hlwhite{Thuật toán}}
\begin{itemize}
\item 3 vòng lặp:
\begin{itemize}
\item $k, i, j$
\end{itemize}
\end{itemize}

\begin{tcolorbox}
\centering
\hlpurple{DP trên tập đỉnh trung gian}
\end{tcolorbox}
\end{frame}

% ===================== CODE =====================

\begin{frame}[fragile]{\hlwhite{Code (Core)}}
\begin{lstlisting}[language=C++]
for(int k = 1; k <= n; k++){
    for(int i = 1; i <= n; i++){
        for(int j = 1; j <= n; j++){
            dist[i][j] = min(
                dist[i][j],
                dist[i][k] + dist[k][j]
            );
        }
    }
}
\end{lstlisting}
\end{frame}

% ===================== ĐỘ PHỨC TẠP =====================

\begin{frame}{\hlwhite{Độ phức tạp}}
\begin{itemize}
\item Floyd:
\[
O(n^3)
\]

\item Dijkstra n lần:
\[
O(n (n + m)\log n)
\]
\end{itemize}

\begin{tcolorbox}
\centering
\hlred{Floyd tốt khi n nhỏ}
\end{tcolorbox}
\end{frame}

% ===================== VẤN ĐỀ =====================

\begin{frame}{\hlwhite{Vấn đề}}
\begin{itemize}
\item \hlred{Negative Cycle}
\begin{itemize}
\item Nếu $dist[i][i] < 0$
\end{itemize}

\item Overflow khi cộng INF
\end{itemize}
\end{frame}

% ===================== KHI NÀO DÙNG =====================

\begin{frame}{\hlwhite{Khi nào dùng}}

\item Cần khoảng cách mọi cặp
\item Đồ thị dày
\end{itemize}
\end{frame}
