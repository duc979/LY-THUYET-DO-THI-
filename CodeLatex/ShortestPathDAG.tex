% ===================== THUẬT TOÁN =====================

\begin{frame}{\hlwhite{Thuật toán}}
\begin{itemize}
\item Topological Sort
\item Khởi tạo:
\begin{itemize}
\item $dist[s] = 0$
\item còn lại = $\infty$
\end{itemize}

\item Duyệt theo topo:
\begin{itemize}
\item Relax các cạnh
\end{itemize}
\end{itemize}
\end{frame}

% ===================== RELAX =====================

\begin{frame}{\hlwhite{Relaxation}}
\begin{tcolorbox}
\centering
\[
dist[v] = \min(dist[v], dist[u] + w(u,v))
\]
\end{tcolorbox}

\begin{itemize}
\item Mỗi cạnh chỉ xét 1 lần
\end{itemize}
\end{frame}

% ===================== CODE =====================

\begin{frame}[fragile]{\hlwhite{Code (Core)}}
\begin{lstlisting}[language=C++]
for(int u : topo){
    for(auto [v, w] : adj[u]){
        if(dist[u] + w < dist[v]){
            dist[v] = dist[u] + w;
        }
    }
}
\end{lstlisting}
\end{frame}

% ===================== ĐỘ PHỨC TẠP =====================

\begin{frame}{\hlwhite{Độ phức tạp}}
\begin{itemize}
\item Topo sort: $O(n + m)$
\item Relax: $O(m)$
\end{itemize}

\begin{tcolorbox}
\centering
\hlred{$O(n + m)$}
\end{tcolorbox}
\end{frame}

% ===================== SO SÁNH =====================

\begin{frame}{\hlwhite{So sánh}}
\begin{itemize}
\item Dijkstra:
\begin{itemize}
\item $O((n+m)\log n)$
\end{itemize}

\item Bellman-Ford:
\begin{itemize}
\item $O(nm)$
\end{itemize}

\item DAG:
\begin{itemize}
\item \hlgreen{$O(n + m)$}
\end{itemize}
\end{itemize}

\hlpurple{Nhanh nhất nếu đồ thị là DAG}

\end{frame}
