% ===================== ĐỀ BÀI =====================

\begin{frame}{\hlwhite{Đề bài}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Bài toán: Biểu diễn đồ thị cơ bản}}]
Cho một đồ thị vô hướng đơn gồm $n$ đỉnh và $m$ cạnh.  
Các đỉnh được đánh số từ $1$ đến $n$.  

Nhiệm vụ của bạn là:
\begin{itemize}
\item In ra \hlblue{danh sách kề} của từng đỉnh theo thứ tự tăng dần
\item In ra \hlblue{ma trận kề} của đồ thị
\end{itemize}
\end{tcolorbox}

\begin{itemize}
\item Đồ thị là \hlgreen{vô hướng}
\item Không có khuyên \hlgray{(self-loop)}
\item Không có nhiều cạnh trùng nhau
\end{itemize}
\end{frame}

% ===================== DIỄN GIẢI YÊU CẦU =====================

\begin{frame}{\hlwhite{Yêu cầu cần làm}}
\begin{tcolorbox}
\centering
\hlpurple{Bài toán kiểm tra xem bạn hiểu đồ thị ở mức cơ bản hay chưa}
\end{tcolorbox}

\begin{itemize}
\item Với mỗi đỉnh $i$, xác định các đỉnh kề với nó
\item Sắp xếp các đỉnh kề theo thứ tự tăng dần
\item Từ các cạnh đã cho, dựng ma trận kề $a[i][j]$
\end{itemize}

\begin{tcolorbox}
\hlred{Nhớ rằng đồ thị vô hướng nên nếu có cạnh } $(u,v)$ \hlred{thì cả } $a[u][v]$ \hlred{và } $a[v][u]$ \hlred{đều bằng 1}
\end{tcolorbox}
\end{frame}

% ===================== DỮ LIỆU VÀO =====================

\begin{frame}{\hlwhite{Dữ liệu vào}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Input}}]
Dòng đầu tiên gồm hai số nguyên $n, m$.

$m$ dòng tiếp theo, mỗi dòng gồm hai số nguyên $u, v$ mô tả một cạnh vô hướng nối giữa hai đỉnh $u$ và $v$.
\end{tcolorbox}

\begin{itemize}
\item $1 \le n \le 1000$
\item $0 \le m \le \dfrac{n(n-1)}{2}$
\item $1 \le u, v \le n$
\item Các cạnh không trùng nhau
\end{itemize}
\end{frame}

% ===================== DỮ LIỆU RA =====================

\begin{frame}{\hlwhite{Dữ liệu ra}}
\begin{tcolorbox}[title=\textbf{\hlwhite{Output}}]
\begin{itemize}
\item In ra danh sách kề của từng đỉnh từ $1$ đến $n$
\item Mỗi dòng ứng với một đỉnh
\item Sau đó in ra ma trận kề của đồ thị
\end{itemize}
\end{tcolorbox}

\begin{itemize}
\item Danh sách kề: các đỉnh kề của một đỉnh trên cùng một dòng
\item Ma trận kề: dòng $i$, cột $j$ bằng:
\[
a[i][j] =
\begin{cases}
1 & \text{nếu có cạnh nối } i \text{ và } j \\
0 & \text{nếu không có cạnh}
\end{cases}
\]
\end{itemize}
\end{frame}

% ===================== GỢI Ý CÁCH LÀM =====================

\begin{frame}{\hlwhite{Gợi ý cách làm}}
\begin{itemize}
\item Dùng \hlblue{vector<int> adj[n+1]} để lưu danh sách kề
\item Dùng mảng \hlgreen{a[n+1][n+1]} để lưu ma trận kề
\item Với mỗi cạnh $(u,v)$:
\begin{itemize}
\item thêm $v$ vào $adj[u]$
\item thêm $u$ vào $adj[v]$
\item gán $a[u][v] = a[v][u] = 1$
\end{itemize}
\item Sau khi đọc hết input, sắp xếp từng danh sách kề
\end{itemize}

\begin{tcolorbox}
\centering
\hlred{Không yêu cầu thuật toán phức tạp, chỉ cần hiểu đúng cách biểu diễn đồ thị}
\end{tcolorbox}
\end{frame}
